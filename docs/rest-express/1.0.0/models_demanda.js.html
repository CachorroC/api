/**
 * @fileoverview Modelo de Demanda Judicial.
 * Transforma datos crudos de la hoja de cálculo (RawDb) en la estructura de demanda
 * utilizada por el sistema, incluyendo medidas cautelares, notificaciones y montos.
 *
 * @module models/demanda
 * @requires prisma/generated/prisma/client
 * @requires types/carpetas
 * @requires types/raw-db
 * @requires utils/capital-builder
 * @requires utils/date-validator
 * @requires models/notificacion
 * @requires models/tipoProceso
 * @package rest-express
 */
import { Prisma } from '../prisma/generated/prisma/client.js';
import { capitalBuilder } from '../utils/capital-builder.js';
import { datesExtractor } from '../utils/date-validator.js';
import { ClassNotificacion } from './notificacion.js';
import { tipoProcesoBuilder } from './tipoProceso.js';
/**
 * @class ClassDemanda
 * @classdesc Representa una demanda judicial dentro del sistema.
 * Encapsula la lógica de transformación de datos crudos (RawDb) a una estructura
 * tipada que incluye obligaciones, fechas procesales, medidas cautelares,
 * notificaciones y montos económicos (capital adeudado, avalúo, liquidación).
 *
 * @implements {IntDemanda}
 * @namespace ClassDemanda
 * @memberof module:models/demanda
 *
 * @example
 * const demanda = new ClassDemanda(rawCarpetaData);
 * console.log(demanda.capitalAdeudado); // Prisma.Decimal
 * console.log(demanda.tipoProceso);     // 'SINGULAR' | 'HIPOTECARIO' | ...
 */
export class ClassDemanda {
    /**
     * Construye una instancia de ClassDemanda a partir de datos crudos.
     *
     * @constructor
     * @param {RawDb} rawCarpeta - Objeto con los datos crudos provenientes de la hoja de cálculo.
     *   Se extraen: capital adeudado, juzgados, fechas de presentación, tipo de proceso,
     *   mandamiento de pago, obligaciones, medidas cautelares, y valores de avalúo/liquidación.
     */
    constructor(rawCarpeta) {
        const { VALOR_CAPITAL_ADEUDADO: capitalAdeudado, JUZGADO_EJECUCION, JUZGADO_ORIGEN, FECHA_ENTREGA_GARANTIAS_ABOGADO: entregaGarantiasAbogado, ETAPA_PROCESAL: etapaProcesal, DEPARTAMENTO: departamento, NUMERO, FECHA_PRESENTACION_DEMANDA: fechaPresentacion, TIPO_PROCESO: tipoProceso, FECHA_MANDAMIENTO_PAGO: mandamientoPago, JUZGADO_CIUDAD: municipio, RADICADO: radicado, EXPEDIENTE: llaveProceso, FECHA_VENCIMIENTO_PAGARE: vencimientoPagare, FECHA_ORDENA_MEDIDAS_CAUTELARES: fechaOrdenaMedidas, MEDIDA_SOLICITADA: medidaSolicitada, OBLIGACION_1: A, OBLIGACION_2: B, VALOR_LIQUIDACION_DEL_CREDITO, VALOR_AVALUO, BIENES, BIENES_SECUESTRADOS, } = rawCarpeta;
        const [newFechaOrdenaMedida] = datesExtractor(fechaOrdenaMedidas);
        this.id = Number(NUMERO);
        this.bienes = BIENES
            ? String(BIENES)
            : BIENES_SECUESTRADOS
                ? String(BIENES_SECUESTRADOS)
                : null;
        this.medidasCautelares = {
            id: Number(NUMERO),
            fechaOrdenaMedida: newFechaOrdenaMedida ?? null,
            medidaSolicitada: medidaSolicitada
                ? String(medidaSolicitada)
                : null,
        };
        const obligacionesSet = new Set();
        if (A) {
            obligacionesSet.add(String(A));
        }
        if (B) {
            obligacionesSet.add(String(B));
        }
        this.fechaPresentacion = datesExtractor(fechaPresentacion) ?? null;
        this.notificacion = new ClassNotificacion(rawCarpeta);
        this.mandamientoPago = datesExtractor(mandamientoPago) ?? null;
        const NewEntregaDeGarantias = datesExtractor(entregaGarantiasAbogado);
        if (NewEntregaDeGarantias.length === 0) {
            this.entregaGarantiasAbogado = null;
        }
        else {
            const [firstEntrega] = NewEntregaDeGarantias;
            this.entregaGarantiasAbogado = firstEntrega;
        }
        this.capitalAdeudado = new Prisma.Decimal(capitalBuilder(capitalAdeudado));
        this.tipoProceso = tipoProcesoBuilder(tipoProceso);
        this.etapaProcesal = etapaProcesal
            ? `${etapaProcesal}`
            : null;
        this.municipio = municipio
            ? String(municipio)
            : null;
        this.obligacion = Array.from(obligacionesSet);
        this.radicado = radicado
            ? `${radicado}`
            : null;
        this.vencimientoPagare = datesExtractor(vencimientoPagare);
        this.departamento = departamento
            ? departamento
            : null;
        this.despacho = JUZGADO_EJECUCION
            ? JUZGADO_EJECUCION
            : JUZGADO_ORIGEN
                ? JUZGADO_ORIGEN
                : null;
        this.llaveProceso = llaveProceso
            ? String(llaveProceso)
            : null;
        this.avaluo = new Prisma.Decimal(capitalBuilder(VALOR_AVALUO));
        this.liquidacion = new Prisma.Decimal(capitalBuilder(VALOR_LIQUIDACION_DEL_CREDITO));
    }
    /** @type {Prisma.Decimal} Valor de liquidación del crédito en moneda colombiana. */
    liquidacion;
    /** @type {Prisma.Decimal} Valor del avalúo del bien asociado. */
    avaluo;
    /** @type {Prisma.Decimal} Capital total adeudado por el demandado. */
    capitalAdeudado;
    /** @type {number} Número de la carpeta asociada a esta demanda. Asignado externamente. */
    carpetaNumero;
    /** @type {string | null} Departamento donde se radicó la demanda. */
    departamento;
    /** @type {string | null} Despacho judicial encargado. */
    despacho;
    /** @type {string | null} Bienes asociados a la demanda. */
    bienes;
    /** @type {Date | null} Fecha de entrega de garantías al abogado. */
    entregaGarantiasAbogado;
    /** @type {string | null} Etapa procesal actual de la demanda. */
    etapaProcesal;
    /** @type {Date[]} Fechas de presentación de la demanda (puede haber múltiples). */
    fechaPresentacion;
    /** @type {number} Identificador numérico único de la demanda. */
    id;
    /** @type {string | null} Llave de proceso (número de radicado de 23 dígitos). */
    llaveProceso;
    /** @type {Date[]} Fechas de mandamiento de pago. */
    mandamientoPago;
    /** @type {string | null} Municipio donde se tramita la demanda. */
    municipio;
    /** @type {intNotificacion} Datos de notificación judicial asociados. */
    notificacion;
    /** @type {string[]} Números de las obligaciones crediticias vinculadas. */
    obligacion;
    /** @type {string | null} Número de radicado del proceso. */
    radicado;
    /** @type {TipoProceso} Tipo de proceso ejecutivo: SINGULAR, HIPOTECARIO, PRENDARIO o ACUMULADO. */
    tipoProceso;
    /** @type {Date[]} Fechas de vencimiento del pagaré. */
    vencimientoPagare;
    /**
     * Medidas cautelares asociadas a la demanda.
     * @type {{id: number, fechaOrdenaMedida: Date|null, medidaSolicitada: string|null}}
     */
    medidasCautelares;
    /**
     * Transforma una instancia de `IntDemanda` en un objeto Prisma `DemandaCreateWithoutCarpetaInput`
     * listo para ser insertado en la base de datos, incluyendo medidas cautelares y notificaciones
     * con lógica `connectOrCreate`.
     *
     * @static
     * @method prismaDemanda
     * @memberof ClassDemanda
     * @param {IntDemanda} demanda - Instancia de demanda con todos sus campos procesados.
     * @returns {Prisma.DemandaCreateWithoutCarpetaInput} Objeto compatible con Prisma para inserción.
     */
    static prismaDemanda(demanda) {
        const newMedidas = {
            id: demanda.id,
            fechaOrdenaMedida: demanda.medidasCautelares.fechaOrdenaMedida,
            medidaSolicitada: demanda.medidasCautelares.medidaSolicitada,
        };
        const newNotificacion = ClassNotificacion.prismaNotificacion(demanda.notificacion);
        const newDemanda = {
            id: demanda.id,
            tipoProceso: demanda.tipoProceso,
            avaluo: demanda.avaluo,
            capitalAdeudado: demanda.capitalAdeudado,
            departamento: demanda.departamento,
            despacho: demanda.despacho,
            entregaGarantiasAbogado: demanda.entregaGarantiasAbogado,
            fechaPresentacion: demanda.fechaPresentacion,
            vencimientoPagare: demanda.vencimientoPagare,
            etapaProcesal: demanda.etapaProcesal,
            liquidacion: demanda.liquidacion,
            llaveProceso: demanda.llaveProceso,
            mandamientoPago: demanda.mandamientoPago,
            municipio: demanda.municipio,
            obligacion: demanda.obligacion,
            radicado: demanda.radicado,
            medidasCautelares: {
                connectOrCreate: {
                    where: {
                        id: demanda.id,
                    },
                    create: newMedidas,
                },
            },
            notificacion: {
                connectOrCreate: {
                    where: {
                        id: demanda.id,
                    },
                    create: newNotificacion,
                },
            },
        };
        return newDemanda;
    }
}
