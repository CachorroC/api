/**
 * @fileoverview Modelo de Procesos Judiciales.
 * Gestiona la consulta de procesos desde la API de la Rama Judicial
 * y la sincronización de sus datos con la base de datos local vía Prisma.
 *
 * @module models/procesos
 * @requires services/prisma
 * @requires types/procesos
 * @requires models/juzgado
 * @package rest-express
 */
import { client } from '../services/prisma';
import JuzgadoClass from './juzgado.js';
/**
 * @class ClassProcesos
 * @classdesc Gestiona la colección de procesos judiciales asociados a una carpeta.
 * Permite consultar la API de la Rama Judicial por llave de proceso,
 * filtrar procesos privados, y sincronizar los resultados con la base de datos
 * mediante operaciones `connectOrCreate` de Prisma.
 *
 * @memberof module:models/procesos
 *
 * @example
 * const procesos = await ClassProcesos.getProcesos('11001400300120180012300', 42, 1);
 * await procesos.prismaUpdateProcesos();
 */
export class ClassProcesos {
    /** @type {Set<number>} Conjunto de IDs de procesos únicos (excluye privados). */
    idProcesosSet = new Set();
    /** @type {outProceso[]} Lista de procesos públicos procesados. */
    procesos = [];
    /** @type {number} Número de la carpeta asociada. */
    numero;
    /** @type {number} ID de la carpeta asociada. */
    carpetaId;
    /**
     * @constructor
     * @param {outProceso[]} procesos - Array de procesos obtenidos de la API.
     * @param {number} numero - Número de carpeta.
     * @param {number} carpetaId - ID de la carpeta.
     */
    constructor(procesos, numero, carpetaId) {
        this.carpetaId = carpetaId;
        this.numero = numero;
        procesos.forEach((proceso) => {
            if (!proceso.esPrivado) {
                this.procesos.push(proceso);
                this.idProcesosSet.add(proceso.idProceso);
            }
        });
    }
    /**
     * Actualiza los procesos de una carpeta en la base de datos.
     * Combina los IDs de procesos existentes con los nuevos y realiza
     * operaciones `connectOrCreate` para procesos y juzgados.
     *
     * @async
     * @method prismaUpdateProcesos
     * @memberof ClassProcesos
     * @returns {Promise<import('../prisma/generated/prisma/client').Carpeta | null>}
     *   La carpeta actualizada o `null` si ocurre un error.
     * @throws {Error} Errores de Prisma se capturan y se registran en consola.
     */
    async prismaUpdateProcesos() {
        try {
            const carpeta = await client.carpeta.findFirstOrThrow({
                where: {
                    numero: this.numero,
                },
            });
            carpeta.idProcesos.forEach((idProceso) => {
                this.idProcesosSet.add(idProceso);
            });
            const updater = await client.carpeta.update({
                where: {
                    numero: this.numero,
                },
                data: {
                    idProcesos: {
                        set: Array.from(this.idProcesosSet),
                    },
                    procesos: {
                        connectOrCreate: this.procesos.map((proceso) => {
                            return {
                                where: {
                                    idProceso: proceso.idProceso,
                                },
                                create: {
                                    ...proceso,
                                    juzgado: {
                                        connectOrCreate: {
                                            where: {
                                                id_tipo_ciudad: {
                                                    id: proceso.juzgado.id,
                                                    tipo: proceso.juzgado.tipo,
                                                    ciudad: proceso.juzgado.ciudad,
                                                },
                                            },
                                            create: proceso.juzgado,
                                        },
                                    },
                                },
                            };
                        }),
                    },
                },
            });
            console.log(updater);
            return updater;
        }
        catch (error) {
            console.log(error);
            return null;
        }
    }
    /**
     * Consulta procesos judiciales desde la API pública de la Rama Judicial
     * usando el número de radicación (llave de proceso).
     * Transforma fechas y extrae datos de juzgado para cada proceso encontrado.
     *
     * @static
     * @async
     * @method getProcesos
     * @memberof ClassProcesos
     * @param {string} llaveProceso - Número de radicado de 23 dígitos.
     * @param {number} [numero=0] - Número de carpeta asociada.
     * @param {number} [carpetaId=0] - ID de la carpeta asociada.
     * @returns {Promise<ClassProcesos>} Instancia con los procesos encontrados.
     *   Retorna instancia vacía si hay error.
     * @throws {Error} Errores HTTP se capturan internamente.
     */
    static async getProcesos(llaveProceso, numero = 0, carpetaId = 0) {
        try {
            const request = await fetch(`https://consultaprocesos.ramajudicial.gov.co:448/api/v2/Procesos/Consulta/NumeroRadicacion?numero=${llaveProceso}&SoloActivos=false&pagina=1`);
            if (!request.ok) {
                throw new Error(`${llaveProceso}: ${request.status} ${request.statusText}${JSON.stringify(request, null, 2)}`);
            }
            const json = (await request.json());
            const { procesos } = json;
            const mappedprocesos = procesos.map((proceso) => {
                return {
                    ...proceso,
                    fechaProceso: proceso.fechaProceso
                        ? new Date(proceso.fechaProceso)
                        : null,
                    fechaUltimaActuacion: proceso.fechaUltimaActuacion
                        ? new Date(proceso.fechaUltimaActuacion)
                        : null,
                    juzgado: JuzgadoClass.fromProceso(proceso),
                };
            });
            return new ClassProcesos(mappedprocesos, numero, carpetaId);
        }
        catch (error) {
            console.log(error);
            return new ClassProcesos([], numero, carpetaId);
        }
    }
}
