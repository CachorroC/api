/**
 * @fileoverview Script de sincronización de carpetas judiciales.
 * Lee datos crudos de la hoja de cálculo, instancia `ClassCarpeta` para cada una,
 * obtiene los procesos asociados desde la API de la Rama Judicial,
 * y persiste todo en la base de datos PostgreSQL.
 *
 * @module services/syncronize_carpetas_test
 * @requires data/carpetas
 * @requires models/carpeta
 * @package rest-express
 */
import { RawCarpetas } from '../data/carpetas.js';
import { ClassCarpeta } from '../models/carpeta.js';
/**
 * Procesa un array de elementos en lotes (batches) de tamaño configurable.
 * Cada lote se ejecuta en paralelo con `Promise.all`, y los lotes se procesan secuencialmente.
 *
 * @async
 * @function processBatch
 * @template T - Tipo de los elementos a procesar.
 * @param {T[]} items - Array completo de items a procesar.
 * @param {number} batchSize - Número de items por lote.
 * @param {(item: T) => Promise<void>} handler - Función asíncrona que procesa cada item.
 * @returns {Promise<void>}
 *
 * @example
 * await processBatch(carpetas, 5, async (carpeta) => {
 *   await carpeta.syncToDatabase();
 * });
 */
async function processBatch(items, batchSize, 
// eslint-disable-next-line no-unused-vars
handler) {
    for (let i = 0; i < items.length; i += batchSize) {
        const batch = items.slice(i, i + batchSize);
        console.log(`Processing batch ${i / batchSize + 1}...`);
        // Run this batch in parallel
        await Promise.all(batch.map((item) => {
            return handler(item);
        }));
    }
}
