/**
 * @fileoverview Detección de patrones importantes en actuaciones judiciales.
 * Analiza textos de actuaciones para detectar palabras clave como "auto" y "radicado"
 * que indican eventos procesales significativos.
 *
 * @module utils/actuacion-has-important-pattern
 * @requires services/prisma
 * @requires types/actuaciones
 * @package rest-express
 */
import { client } from '../services/prisma.js';
import * as fs from 'fs/promises';
/**
 * Obtiene todas las actuaciones de la base de datos, les aplica detección
 * de patrones "auto" y "radicado", y escribe los resultados en un archivo JSON.
 *
 * @async
 * @function actuacionesGetAuto
 * @memberof module:utils/actuacion-has-important-pattern
 * @returns {Promise<Array<intActuacion & {hasAuto: boolean, hasRadicado: boolean}>>}
 *   Array de actuaciones enriquecidas con flags de patrones detectados.
 */
export async function actuacionesGetAuto() {
    const actuaciones = await client.actuacion.findMany();
    const newActs = actuaciones.map((actuacion) => {
        return actuacionHasAuto(actuacion);
    });
    fs.writeFile('actuacionesNewMap.json', JSON.stringify(newActs));
    return newActs;
}
/**
 * Detecta si una actuación contiene las palabras clave "auto" o "radicado".
 * Utiliza expresiones regulares case-insensitive para la detección.
 *
 * @function actuacionHasAuto
 * @memberof module:utils/actuacion-has-important-pattern
 * @param {intActuacion} incomingActuacion - La actuación a analizar.
 * @returns {intActuacion & {hasAuto: boolean, hasRadicado: boolean}}
 *   La actuación original enriquecida con los flags `hasAuto` y `hasRadicado`.
 *
 * @example
 * const result = actuacionHasAuto(actuacion);
 * if (result.hasAuto) console.log('Contiene AUTO');
 */
export function actuacionHasAuto(incomingActuacion) {
    const { actuacion } = incomingActuacion;
    const hasAuto = /([Aa][Uu][Tt][Oo]+)\s([A-Z a-z����������,.()\w/]+)/.test(actuacion);
    const hasRadicado = /([Rr][Aa][Dd][Ii][Cc][Aa]+)([A-Z a-z����������,.()\w/]+)/.test(actuacion);
    return {
        ...incomingActuacion,
        hasAuto: hasAuto,
        hasRadicado: hasRadicado,
    };
}
