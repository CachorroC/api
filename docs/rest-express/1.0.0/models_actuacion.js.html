/**
 * @fileoverview Modelo de Actuación Judicial.
 * Provee métodos estáticos para la sincronización y persistencia de actuaciones
 * judiciales obtenidas de la API de la Rama Judicial de Colombia.
 *
 * @module models/actuacion
 * @requires fs/promises
 * @requires services/prisma
 * @requires types/actuaciones
 * @see {@link https://consultaprocesos.ramajudicial.gov.co:448 Rama Judicial API}
 * @package rest-express
 */
import * as fs from 'fs/promises';
import { client } from '../services/prisma.js';
/**
 * @class Actuacion
 * @classdesc Clase estática que encapsula toda la lógica de persistencia y sincronización
 * de actuaciones judiciales con la base de datos PostgreSQL a través de Prisma ORM.
 *
 * No se instancia directamente; todos sus métodos son `static`.
 *
 * @namespace Actuacion
 * @memberof module:models/actuacion
 *
 * @example
 * // Actualizar actuaciones para la carpeta 123
 * const actuaciones = await fetchActuaciones(idProceso);
 * await Actuacion.updateAllActuaciones(actuaciones, 123);
 *
 * @example
 * // Sincronizar la última actuación de una carpeta
 * await Actuacion.prismaUpdaterActuaciones(actuaciones, 123, 456);
 */
export default class Actuacion {
    /**
     * Realiza upsert (inserción o actualización) de todas las actuaciones proporcionadas
     * en la base de datos, vinculándolas a una carpeta específica.
     *
     * @static
     * @async
     * @method updateAllActuaciones
     * @memberof Actuacion
     * @param {outActuacion[]} actuacionesComplete - Array completo de actuaciones obtenidas de la API.
     * @param {number} numeroCarpeta - Número identificador de la carpeta a la que pertenecen las actuaciones.
     * @returns {Promise<Array<{idRegActuacion: string, actuacion: string, anotacion: string|null, cant: number, carpetaNumero: number|null, codRegla: string, conDocumentos: boolean, consActuacion: number, createdAt: Date, fechaActuacion: Date, fechaFinal: Date|null, fechaInicial: Date|null, fechaRegistro: Date, isUltimaAct: boolean, llaveProceso: string, procesoId: number|null, idProceso: number}>>} Array de actuaciones upsertadas exitosamente.
     * @throws {Error} Errores de Prisma se capturan internamente y se registran en consola.
     */
    static async updateAllActuaciones(actuacionesComplete, numeroCarpeta) {
        const outputActuaciones = new Set();
        try {
            for (const actuacion of actuacionesComplete) {
                const upsertActuacion = await client.actuacion.upsert({
                    where: {
                        idRegActuacion: `${actuacion.idRegActuacion}`,
                    },
                    update: {
                        ...actuacion,
                        idRegActuacion: `${actuacion.idRegActuacion}`,
                        idProceso: Number(actuacion.idProceso),
                        carpetaNumero: numeroCarpeta,
                        isUltimaAct: actuacion.consActuacion === actuacion.cant,
                        Carpeta: {
                            connect: {
                                numero: numeroCarpeta,
                            },
                        },
                        proceso: {
                            connect: {
                                idProceso: actuacion.idProceso,
                            },
                        },
                    },
                    create: {
                        ...actuacion,
                        idRegActuacion: `${actuacion.idRegActuacion}`,
                        idProceso: Number(actuacion.idProceso),
                        carpetaNumero: numeroCarpeta,
                        isUltimaAct: actuacion.consActuacion === actuacion.cant,
                        Carpeta: {
                            connect: {
                                numero: numeroCarpeta,
                            },
                        },
                        proceso: {
                            connect: {
                                idProceso: actuacion.idProceso,
                            },
                        },
                    },
                });
                console.log(JSON.stringify(upsertActuacion));
                outputActuaciones.add(upsertActuacion);
            }
        }
        catch (error) {
            console.log(error);
        }
        return Array.from(outputActuaciones);
    }

     * 4. Escribe un archivo JSON de respaldo en `./src/date/{año}/{mes}/{día}/`.
     *
     * @static
     * @async
     * @method prismaUpdaterActuaciones

     * @param {outActuacion[]} actuacionesComplete - Array de actuaciones procesadas.

     * @param {number} numeroId - ID alternativo de la carpeta (usado como fallback en búsqueda).

    /**
     * Actualiza una carpeta judicial en la base de datos conectando la nueva última actuación.
     * Utiliza `connectOrCreate` para asegurar que la actuación exista antes de vincularla.
     *
     * @static
     * @async
     * @method updateCarpetaWithNewLastActuacion
     * @memberof Actuacion
     * @param {Object} params - Parámetros de actualización.
     * @param {outActuacion} params.ultimaActuacion - La actuación más reciente a vincular.
     * @param {number} params.numero - Número de la carpeta a actualizar.
     * @returns {Promise<import('../prisma/generated/prisma/client').Carpeta | null>} La carpeta actualizada con su última actuación, o `null` si ocurre un error.
     * @throws {Error} Errores de Prisma se capturan y se retorna `null`.
     */
    static async updateCarpetaWithNewLastActuacion({ ultimaActuacion, numero, }) {
        try {
            const carpetaUpdate = await client.carpeta.update({
                where: {
                    numero: numero,
                },
                data: {
                    fecha: new Date(ultimaActuacion.fechaActuacion),
                    revisado: false,
                    ultimaActuacion: {
                        connectOrCreate: {
                            where: {
                                idRegActuacion: `${ultimaActuacion.idRegActuacion}`,
                            },
                            create: {
                                ...ultimaActuacion,
                                isUltimaAct: ultimaActuacion.cant === ultimaActuacion.consActuacion,
                                idRegActuacion: `${ultimaActuacion.idRegActuacion}`,
                                idProceso: Number(ultimaActuacion.idProceso),
                                carpetaNumero: numero,
                                proceso: {
                                    connect: {
                                        idProceso: ultimaActuacion.idProceso,
                                    },
                                },
                            },
                        },
                    },
                },
                include: {
                    ultimaActuacion: true,
                },
            });
            return carpetaUpdate;
        }
        catch (error) {
            console.log(`error al cambiar la carpeta y actualizar la ultima actuacion: ${error}`);
            return null;
        }
    }
    /**
     * Marca la actuación previamente señalada como "última" (`isUltimaAct = true`)
     * como no última (`isUltimaAct = false`), permitiendo que una nueva tome su lugar.
     *
     * @static
     * @async
     * @method updatePreviousLastActuacion
     * @memberof Actuacion
     * @param {string | null} idRegUltimaAct - El ID de registro de la actuación anteriormente marcada como última.
     * @returns {Promise<import('../prisma/generated/prisma/client').Actuacion | null>} La actuación actualizada, o `null` si el ID es nulo o hay error.
     * @throws {Error} Lanza error interno si `idRegUltimaAct` es `null`; errores de Prisma se capturan.
     */
    static async updatePreviousLastActuacion(idRegUltimaAct) {
        try {
            if (idRegUltimaAct === null) {
                throw new Error('la idRegAct de la ultima actuacion es null');
            }
            const updateCarpeta = await client.actuacion.update({
                where: {
                    idRegActuacion: idRegUltimaAct,
                },
                data: {
                    isUltimaAct: false,
                },
            });
            return updateCarpeta;
        }
        catch (error) {
            console.log(`error al cambiar la ultima actuacion: ${error}`);
            return null;
        }
    }
}
