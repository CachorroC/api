/**
 * @fileoverview Servicio de sincronización masiva de actuaciones judiciales.
 * Combina la obtención de procesos desde la base de datos con la consulta
 * de actuaciones a la API de la Rama Judicial, usando rate limiting.
 *
 * @module services/actuaciones-raw
 * @requires fs/promises
 * @requires services/prisma
 * @requires services/fetch-actuaciones
 * @requires async-sema
 * @package rest-express
 */
import * as fs from 'fs/promises';
import { client } from './prisma.js';
import fetchActuaciones from './fetch-actuaciones.js';
import { RateLimit } from 'async-sema';
process.env['NODE_TLS_REJECT_UNAUTHORIZED'] = '0';
console.log(process.env.NODE_TLS_REJECT_UNAUTHORIZED);
/**
 * Obtiene todos los IDs de procesos desde la base de datos,
 * aplanando los arrays `idProcesos` de cada carpeta.
 *
 * @async
 * @function getIdProcesos
 * @memberof module:services/actuaciones-raw
 * @returns {Promise<Array<{idProceso: number, numero: number}>>}
 *   Array de objetos con el ID del proceso y el número de carpeta.
 */
async function getIdProcesos() {
    const carpetas = await client.carpeta.findMany();
    return carpetas.flatMap((carpeta) => {
        return carpeta.idProcesos.map((idProceso) => {
            return {
                idProceso,
                numero: carpeta.numero,
            };
        });
    });
}
/**
 * Generador asíncrono que itera sobre procesos y obtiene sus actuaciones.
 * Yield de un objeto con el resultado del fetch y el número de carpeta.
 *
 * @async
 * @generator
 * @function AsyncGenerateActuaciones
 * @memberof module:services/actuaciones-raw
 * @param {Array<{idProceso: number, numero: number}>} procesos - Lista de procesos a consultar.
 * @yields {{fetchBody: Awaited<ReturnType<typeof fetchActuaciones>>, numero: number}}
 *   Resultado de cada consulta de actuaciones con su número de carpeta.
 */
async function* AsyncGenerateActuaciones(procesos) {
    for (const { idProceso, numero } of procesos) {
        console.log(numero);
        const fetcherIdProceso = await fetchActuaciones(idProceso);
        yield {
            fetchBody: fetcherIdProceso,
            numero: numero,
        };
    }
}

/**
 * Función principal de sincronización masiva.
 * Obtiene todos los procesos, consulta sus actuaciones con rate limiting (5 req/s),
 * y escribe los resultados en `actuacionesFullOutput.json`.
 *
 * @async
 * @function main
 * @memberof module:services/actuaciones-raw
 * @returns {Promise<Array<Awaited<ReturnType<typeof fetchActuaciones>>>>}
 *   Array completo de todas las actuaciones obtenidas.
 */
async function main() {
    const ActsMap = [];
    const idProcesos = await getIdProcesos();
    const lim = RateLimit(5);
    for (const { idProceso, numero } of idProcesos) {
        await lim();
        const acts = await fetchActuaciones(idProceso, numero);
        ActsMap.push(acts);
    }
    fs.writeFile('actuacionesFullOutput.json', JSON.stringify(ActsMap, null, 2));
    return ActsMap;
}
main();
