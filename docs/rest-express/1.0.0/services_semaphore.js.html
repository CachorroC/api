#!/usr/bin/env node
/**
 * @fileoverview Demostración de rate limiting con async-sema.
 * Ejecuta tres escenarios de limitación de peticiones por segundo/minuto
 * para validar el comportamiento del módulo `async-sema`.
 *
 * @module services/semaphore
 * @requires async-sema
 * @package rest-express
 */
import { RateLimit } from 'async-sema';
/**
 * Demuestra rate limiting básico: 5 peticiones por segundo.
 * Mide el throughput real en req/s.
 *
 * @async
 * @function f
 * @memberof module:services/semaphore
 * @returns {Promise<void>}
 */
async function f() {
    console.log('Naive requests per second rate limiting');
    const n = 50;
    const lim = RateLimit(5);
    const start = process.hrtime();
    for (let i = 0; i < n; i++) {
        await lim();
        process.stdout.write('.');
    }
    process.stdout.write('\n');
    const hrt = process.hrtime(start);
    const elapsed = (hrt[0] * 1000 + hrt[1] / 1e6) / 1000;
    const rps = n / elapsed;
    console.log(rps.toFixed(3) + ' req/s');
}
/**
 * Demuestra rate limiting con unidad de tiempo personalizada (por minuto).
 * Configura el límite a 5 peticiones por cada 60,000ms.
 *
 * @async
 * @function g
 * @memberof module:services/semaphore
 * @returns {Promise<void>}
 */
async function g() {
    console.log('Custom rate limit time unit');
    const n = 20;
    const lim = RateLimit(5, {
        timeUnit: 60 * 1000,
    });
    const start = process.hrtime();
    for (let i = 0; i < n; i++) {
        await lim();
        process.stdout.write('.');
    }
    process.stdout.write('\n');
    const hrt = process.hrtime(start);
    const elapsed = (hrt[0] * 1000 + hrt[1] / 1e6) / 1000;
    const rps = n / (elapsed / 60);
    console.log(rps.toFixed(3) + ' req/min');
}
/**
 * Demuestra rate limiting con distribución uniforme.
 * Las peticiones se espacian equitativamente en el tiempo.
 *
 * @async
 * @function h
 * @memberof module:services/semaphore
 * @returns {Promise<void>}
 */
async function h() {
    console.log('Uniform distribution of requests over time');
    const n = 50;
    const lim = RateLimit(5, {
        uniformDistribution: true,
    });
    const start = process.hrtime();
    for (let i = 0; i < n; i++) {
        await lim();
        process.stdout.write('.');
    }
    process.stdout.write('\n');
    const hrt = process.hrtime(start);
    const elapsed = (hrt[0] * 1000 + hrt[1] / 1e6) / 1000;
    const rps = n / elapsed;
    console.log(rps.toFixed(3) + ' req/s');
}
f()
    .then(g)
    .then(h)
    .catch((e) => {
    return console.log(e);
})
    .then(() => {
    return console.log('READY');
});
