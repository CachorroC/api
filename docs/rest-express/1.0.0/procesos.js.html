/**
 * @fileoverview Script de sincronización de procesos judiciales.
 * Consulta la API de Rama Judicial por llave de proceso,
 * normaliza los datos (fechas, juzgados) y los persiste en PostgreSQL
 * usando `connectOrCreate` para juzgados y procesos.
 *
 * @module src/procesos
 * @requires types/procesos
 * @requires models/juzgado
 * @requires models/procesos
 * @requires services/prisma
 * @package rest-express
 */
import * as fs from 'fs/promises';
import JuzgadoClass from './models/juzgado.js';
import { ClassProcesos } from './models/procesos.js';
import { client } from './services/prisma.js';
/**
 * Consulta procesos judiciales por llave en la API de Rama Judicial
 * y normaliza las fechas y juzgados de cada proceso encontrado.
 *
 * @async
 * @function fetcher
 * @memberof module:src/procesos
 * @param {string} llaveProceso - Llave de 23 dígitos del proceso.
 * @returns {Promise<outProceso[]>} Array de procesos normalizados. Vacío si falla.
 * @throws {Error} Si la respuesta HTTP no es exitosa.
 */
async function fetcher(llaveProceso) {
    try {
        const request = await fetch(`https://consultaprocesos.ramajudicial.gov.co:448/api/v2/Procesos/Consulta/NumeroRadicacion?numero=${llaveProceso}&SoloActivos=false&pagina=1`);
        if (!request.ok) {
            throw new Error(`${llaveProceso}: ${request.status} ${request.statusText}${JSON.stringify(request, null, 2)}`);
        }
        const json = (await request.json());
        const { procesos } = json;
        return procesos.map((proceso) => {
            return {
                ...proceso,
                fechaProceso: proceso.fechaProceso
                    ? new Date(proceso.fechaProceso)
                    : null,
                fechaUltimaActuacion: proceso.fechaUltimaActuacion
                    ? new Date(proceso.fechaUltimaActuacion)
                    : null,
                juzgado: JuzgadoClass.fromProceso(proceso),
            };
        });
    }
    catch (error) {
        console.log(error);
        return [];
    }
}
/**
 * Obtiene todas las llaves de proceso de las carpetas en base de datos.
 *
 * @async
 * @function getLLaves
 * @memberof module:src/procesos
 * @returns {Promise<Array<{llaveProceso: string, numero: number, id: number}>>}
 *   Array con llaveProceso y metadatos de cada carpeta.
 */
async function getLLaves() {
    const carpetas = await client.carpeta.findMany();
    return carpetas.flatMap((carpeta) => {
        return {
            llaveProceso: carpeta.llaveProceso.trim(),
            numero: carpeta.numero,
            id: carpeta.id,
        };
    });
}
/**
 * Generador asíncrono que itera carpetas, consulta procesos y los persiste en BD.
 *
 * @async
 * @generator
 * @function AsyncGenerateActuaciones
 * @memberof module:src/procesos
 * @param {Array<{llaveProceso: string, id: number, numero: number}>} llaves - Lista de llaves.
 * @yields {ClassProcesos} Instancia con procesos actualizados para cada carpeta.
 */
async function* AsyncGenerateActuaciones(llaves) {
    for (const carpeta of llaves) {
        const newProceso = await ClassProcesos.getProcesos(carpeta.llaveProceso, carpeta.numero);
        const fetcherIdProceso = await fetcher(carpeta.llaveProceso);
        for (const proceso of fetcherIdProceso) {
            if (!proceso.esPrivado) {
                await prismaUpdaterProcesos(proceso, carpeta.numero);
            }
        }
        newProceso.prismaUpdateProcesos();
        yield newProceso;
    }
}
/**
 * Persiste un proceso en PostgreSQL usando `connectOrCreate` para el juzgado.
 * Actualiza el array `idProcesos` de la carpeta con el nuevo ID.
 *
 * @async
 * @function prismaUpdaterProcesos
 * @memberof module:src/procesos
 * @param {outProceso} proceso - Proceso normalizado a persistir.
 * @param {number} numero - Número de carpeta asociada.
 * @returns {Promise<void>}
 */
async function prismaUpdaterProcesos(proceso, numero) {
    const idProcesosSet = new Set();
    try {
        const carpeta = await client.carpeta.findFirstOrThrow({
            where: {
                numero: numero,
            },
        });
        carpeta.idProcesos.forEach((idProceso) => {
            idProcesosSet.add(idProceso);
        });
        idProcesosSet.add(proceso.idProceso);
        const updater = await client.carpeta.update({
            where: {
                numero: numero,
            },
            data: {
                idProcesos: {
                    set: Array.from(idProcesosSet),
                },
                procesos: {
                    connectOrCreate: {
                        where: {
                            idProceso: proceso.idProceso,
                        },
                        create: {
                            ...proceso,
                            juzgado: {
                                connectOrCreate: {
                                    where: {
                                        id_tipo_ciudad: {
                                            tipo: proceso.juzgado.tipo,
                                            id: proceso.juzgado.id,
                                            ciudad: proceso.juzgado.ciudad,
                                        },
                                    },
                                    create: {
                                        tipo: proceso.juzgado.tipo,
                                        id: proceso.juzgado.id,
                                        ciudad: proceso.juzgado.ciudad,
                                        url: proceso.juzgado.url,
                                    },
                                },
                            },
                        },
                    },
                },
            },
        });
        console.log(updater);
    }
    catch (error) {
        console.log(error);
    }
}
/**
 * Punto de entrada: sincroniza todos los procesos de todas las carpetas.
 * Itera sobre todas las llaves, consulta la API, y escribe el resultado a JSON.
 *
 * @async
 * @function main
 * @memberof module:src/procesos
 * @returns {Promise<ClassProcesos[]>} Array de todos los procesos sincronizados.
 */
async function main() {
    const ActsMap = [];
    const idProcesos = await getLLaves();
    console.log(idProcesos);
    for await (const actuacionesJson of AsyncGenerateActuaciones(idProcesos)) {
        console.log(actuacionesJson);
        ActsMap.push(actuacionesJson);
    }
    fs.writeFile('actuacionesOutput.json', JSON.stringify(ActsMap));
    return ActsMap;
}
main();
