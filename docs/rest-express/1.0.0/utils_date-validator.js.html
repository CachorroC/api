/**
 * @fileoverview Utilidades de validación y extracción de fechas.
 * Provee funciones para parsear fechas desde múltiples formatos de entrada
 * (strings, números de Excel, objetos Date) y validarlas dentro de rangos razonables.
 *
 * @module utils/date-validator
 * @package rest-express
 */
/* eslint-disable no-unused-vars */
/**
 * Extrae y parsea fechas desde múltiples formatos de entrada.
 * Soporta:
 * - Objetos `Date` directos
 * - Números de serie de Excel (vía `xlsxNumberToDate`)
 * - Strings con fechas separadas por `//` (múltiples fechas)
 * - Strings con formato `DD/MM/YYYY` o `DD-MM-YYYY`
 *
 * @function datesExtractor
 * @memberof module:utils/date-validator
 * @param {string | number | Date | null} [incomingDate] - Valor de fecha en formato variable.
 * @returns {Date[]} Array de fechas válidas extraídas. Array vacío si no se encontraron.
 *
 * @example
 * datesExtractor('15/03/2025');           // [Date('2025-03-15')]
 * datesExtractor('10/01/2020//20/05/2021'); // [Date, Date]
 * datesExtractor(44197);                  // [Date] (número de Excel)
 * datesExtractor(null);                   // []
 */
export function datesExtractor(incomingDate) {
    const outputDates = [];
    if (!incomingDate) {
        return dateArrayValidator(outputDates);
    }
    if (typeof incomingDate === 'object') {
        //console.log( incomingDate );
        if (incomingDate.toString() !== 'Invalid Date') {
            outputDates.push(incomingDate);
        }
        return dateArrayValidator(outputDates);
    }
    if (typeof incomingDate === 'number') {
        const outgoingDate = xlsxNumberToDate(incomingDate);
        if (outgoingDate) {
            outputDates.push(outgoingDate);
        }
        return dateArrayValidator(outputDates);
    }
    const splitByDoubleSlash = incomingDate.split('//');
    for (const splitted of splitByDoubleSlash) {
        const fixed = fixSingleFecha(splitted);
        for (const fixedDate of fixed) {
            outputDates.push(fixedDate);
        }
    }
    return dateArrayValidator(outputDates);
}
/**
 * Parsea una cadena de texto individual que contiene una o más fechas en formato
 * `DD/MM/YYYY` o `DD-MM-YYYY`. Usa regex global para detectar todas las ocurrencias.
 *
 * @function fixSingleFecha
 * @memberof module:utils/date-validator
 * @param {string} rawFecha - Cadena con potenciales fechas.
 * @returns {Date[]} Array de fechas parseadas desde la cadena.
 *
 * @example
 * fixSingleFecha('Presentada 15/03/2025'); // [Date('2025-03-15')]
 */
export function fixSingleFecha(rawFecha) {
    const datesOutput = [];
    const matchedDate = rawFecha.matchAll(/(\d+)(-|\/)(\d+)(-|\/)(\d+)/gm);
    for (const matchedValue of matchedDate) {
        const [total, firstNumber, firstDivider, secondNumber, secondDivider, thirdNumber,] = matchedValue;
        const newMonth = Number(secondNumber) - 1;
        const newYear = Number(thirdNumber.padStart(4, '2015'));
        const newDay = Number(firstNumber.padStart(2, '00'));
        const outputDate = new Date(newYear, newMonth, newDay);
        datesOutput.push(outputDate);
    }
    return datesOutput;
}
/**
 * Valida que una fecha sea razonable: no sea "Invalid Date",
 * su año sea mayor a 2000....
 *
 * @function dateValidator
 * @memberof module:utils/date-validator
 * @param {Date} incomingDate - Fecha a validar.
 * @returns {Date | null} La fecha si es válida, o `null` si no pasa la validación.
 */
//!SECTION
export function dateValidator(incomingDate) {
    const stringifiedDate = incomingDate.toString();
    //console.log( stringifiedDate );
    /*console.log( `
      string date:
      ${ String( incomingDate ) }
      ` );*/
    const dateYear = incomingDate.getFullYear();
    //console.log( dateYear );
    if (stringifiedDate === 'Invalid Date'
        || dateYear <= 2000
        || dateYear > 2200) {
        //console.log( dateYear );
        return null;
    }
    //console.log( incomingDate );
    return incomingDate;
}
/**
 * Valida un array de fechas, eliminando las que no pasan `dateValidator`.
 *
 * @function dateArrayValidator
 * @memberof module:utils/date-validator
 * @param {Date[]} incomingDates - Array de fechas a validar.
 * @returns {Date[]} Solo las fechas que pasaron la validación.
 */
export function dateArrayValidator(incomingDates) {
    const outPutDates = [];
    for (const date of incomingDates) {
        const ndate = dateValidator(date);
        if (ndate) {
            outPutDates.push(ndate);
        }
    }
    return outPutDates;
}
/**
 * Convierte un número serial de Excel a un objeto `Date`.
 * Excel almacena fechas como días desde el 30/12/1899.
 * Rechaza valores mayores a 55000 (~2050) o conversiones inválidas.
 *
 * @function xlsxNumberToDate
 * @memberof module:utils/date-validator
 * @param {number} incomingDate - Número serial de Excel.
 * @returns {Date | null} Fecha convertida, o `null` si el número está fuera de rango.
 *
 * @example
 * xlsxNumberToDate(44197); // Date('2021-01-01')
 * xlsxNumberToDate(99999); // null
 */
export function xlsxNumberToDate(incomingDate) {
    const outgoingDate = new Date((incomingDate - (25567 + 1)) * 86400 * 1000);
    if (incomingDate > 55000
        || outgoingDate.toString() === 'Invalid Date'
        || outgoingDate.getFullYear() > 2200) {
        console.log(outgoingDate.toString());
        return null;
    }
    return outgoingDate;
}
