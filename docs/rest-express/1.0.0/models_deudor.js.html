/**
 * @class Tel
 * @classdesc Extrae y separa números telefónicos fijos (7 dígitos) y celulares (10 dígitos)
 * a partir de una cadena de texto libre.
 *
 * @memberof module:models/deudor
 *
 * @example
 * const tel = new Tel('3101234567 2345678');
 * console.log(tel.celular); // '3101234567'
 * console.log(tel.fijo);    // '2345678'
 */
export class Tel {
    /** @type {string | null} Número de teléfono fijo (7 dígitos), o `null` si no se encontró. */
    fijo;
    /** @type {string | null} Número de teléfono celular (10 dígitos), o `null` si no se encontró. */
    celular;
    /**
     * @constructor
     * @param {string} telefono - Cadena de texto con uno o más números telefónicos.
     */
    constructor(telefono) {
        const celularStringArray = telefono.match(/\d{10}/g);
        const fijoStringArray = telefono.match(/\d{7}\s/g);
        const celularNumber = celularStringArray?.map((f) => {
            return String(f);
        });
        const fijoNumber = fijoStringArray?.map((f) => {
            return String(f);
        });
        this.fijo = fijoNumber
            ? fijoNumber[0]
            : null;
        this.celular = celularNumber
            ? celularNumber[0]
            : null;
    }
}
/**
 * @class ClassDeudor
 * @classdesc Representa al deudor (demandado) de un proceso judicial.
 * Extrae y normaliza datos personales desde los datos crudos de la hoja de cálculo,
 * incluyendo separación inteligente del nombre completo en sus componentes
 * (primer nombre, segundo nombre, primer apellido, segundo apellido)
 * según la cantidad de palabras detectadas.
 *
 * @implements {IntDeudor}
 * @memberof module:models/deudor
 *
 * @example
 * const deudor = new ClassDeudor(rawCarpetaData);
 * console.log(deudor.primerNombre);  // 'JUAN'
 * console.log(deudor.primerApellido); // 'PEREZ'
 */
export class ClassDeudor {
    /**
     * @constructor
     * @param {RawDb} rawCarpeta - Datos crudos del deudor desde la fuente original.
     *   Se extraen: identificación, dirección, email, teléfonos, y nombre completo.
     */
    constructor(rawCarpeta) {
        const { DEMANDADO_IDENTIFICACION: cedula, DEMANDADO_DIRECCION: direccion, DEMANDADO_EMAIL: email, DEMANDADO_TELEFONOS: telefono, DEMANDADO_NOMBRE: nombre, NUMERO: id, } = rawCarpeta;
        this.id = Number(id);
        this.cedula = String(cedula);
        this.direccion = direccion
            ? direccion.toString()
            : null;
        this.email = email
            ? email.toString()
            : null;
        const { fijo, celular } = new Tel(String(telefono));
        this.telCelular = celular;
        this.telFijo = fijo;
        const nameStringArray = nombre
            ? nombre.trim()
                .split(' ')
            : 'Nelson Nu�ez'.split(' ');
        const nameArrayLength = nameStringArray.length;
        switch (nameArrayLength) {
            case 4:
                [
                    this.primerNombre,
                    this.segundoNombre,
                    this.primerApellido,
                    this.segundoApellido,
                ] = nameStringArray;
                break;
            case 2:
                [
                    this.primerNombre,
                    this.primerApellido
                ] = nameStringArray;
                this.segundoApellido = null;
                this.segundoNombre = null;
                break;
            case 1:
                [
                    this.primerNombre
                ] = nameStringArray;
                this.primerApellido = 'sinEspecificar';
                this.segundoApellido = null;
                this.segundoNombre = null;
                break;
            case 3:
                [
                    this.primerNombre,
                    this.segundoNombre,
                    this.primerApellido
                ]
                    = nameStringArray;
                this.segundoApellido = null;
                break;
            default:
                [
                    this.primerNombre,
                    this.segundoNombre,
                    this.primerApellido,
                    this.segundoApellido,
                ] = nameStringArray;
                break;
        }
    }
    /** @type {number} Identificador numérico único del deudor. */
    id;
    /** @type {string | null} Número de teléfono celular. */
    telCelular;
    /** @type {string | null} Número de teléfono fijo. */
    telFijo;
    /** @type {string} Primer nombre del deudor. */
    primerNombre;
    /** @type {string | null} Segundo nombre del deudor, si tiene. */
    segundoNombre;
    /** @type {string} Primer apellido del deudor. */
    primerApellido;
    /** @type {string | null} Segundo apellido del deudor, si tiene. */
    segundoApellido;
    /** @type {string} Cédula de ciudadanía del deudor. */
    cedula;
    /** @type {string | null} Dirección de residencia del deudor. */
    direccion;
    /** @type {string | null} Correo electrónico del deudor. */
    email;
    /**
     * Transforma una instancia `IntDeudor` en un objeto Prisma `DeudorCreateWithoutCarpetaInput`
     * listo para inserción en la base de datos.
     *
     * @static
     * @method prismaDeudor
     * @memberof ClassDeudor
     * @param {IntDeudor} deudor - Instancia del deudor con campos procesados.
     * @returns {Prisma.DeudorCreateWithoutCarpetaInput} Objeto compatible con Prisma.
     */
    static prismaDeudor(deudor) {
        const newDeudor = {
            id: deudor.id,
            cedula: deudor.cedula,
            primerApellido: deudor.primerApellido,
            primerNombre: deudor.primerNombre,
            direccion: deudor.direccion,
            email: deudor.email,
            segundoApellido: deudor.segundoApellido,
            segundoNombre: deudor.segundoNombre,
            telCelular: deudor.telCelular,
            telFijo: deudor.telFijo,
        };
        return newDeudor;
    }
}
